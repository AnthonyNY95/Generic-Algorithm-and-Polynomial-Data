import numpy.random as rand
import numpy as np
from copy import deepcopy
import matplotlib.pyplot as plt

def start_population(p):
    chromosomes = []
    for i in range(p): 
        chromosomes.append([random.randint(-100,100) for i in range(4)])
    return chromosomes 

def mutation(person): 
    mutations = []
    mutations.append(person)
    for _ in range(20): 
        new = deepcopy(person)
        rand_idx = random.randrange(len(new))
        rand_num = new[rand_idx]
        rand_float = random.uniform(0.0,4.0)
        
        new_num = rand_num * rand_float
        new[rand_idx] = new_num
        
        mutations.append(new)
    return mutations

def crossover(p1, p2): 
    c1, c2 = [], []
    rand_idx = random.randrange(len(p1))
    c1 = p1[:rand_idx]+ p2[rand_idx:]
    c2 = p2[:rand_idx] + p1[rand_idx:]
    
    return c1, c2

def GA(): 
    num_gens = 1000
    genes = []
    
    population = start_population(5000)
    for i in range(num_gens): 
        new_pop = []
        genes = pop_fitness(population)
        
        cross = []
        for k in range(0, len(genes)-1,2):  
            p1, p2 = crossover(genes[k][0], genes[k+1][0])
            cross.append(p1)
            cross.append(p2)
            
        for b in range(len(cross)): 
            newGenes = mutation(cross[b])
             
            for j in newGenes: 
                new_pop.append(j)
        
        population = deepcopy(new_pop)

    return population
